# Célula de Código
import matplotlib.pyplot as plt
import numpy as np

# Célula de Markdown
"""
---

### Exercício 1 (Físico-Química): Curva de Titulação

**Cenário:** Numa titulação ácido-base, o pH da solução varia à medida que o volume de titulante adicionado aumenta.

**Dados:**
* `volume_titulante_ml = np.array([0, 5, 10, 15, 20, 22, 23, 24, 25, 26, 27, 28, 30, 35, 40])`
* `ph_solucao = np.array([2.0, 2.2, 2.5, 3.0, 4.0, 5.0, 6.0, 7.0, 8.0, 9.0, 10.0, 11.0, 11.5, 11.8, 12.0])`

**Tarefa:**
1.  Crie um gráfico de linhas que represente a curva de titulação (pH vs. volume de titulante).
2.  Adicione um título apropriado ao gráfico.
3.  Adicione rótulos aos eixos X e Y.
4.  Adicione uma grelha ao gráfico.
"""

# Célula de Código (para o Exercício 1)
volume_titulante_ml = np.array([0, 5, 10, 15, 20, 22, 23, 24, 25, 26, 27, 28, 30, 35, 40])
ph_solucao = np.array([2.0, 2.2, 2.5, 3.0, 4.0, 5.0, 6.0, 7.0, 8.0, 9.0, 10.0, 11.0, 11.5, 11.8, 12.0])

# Célula de Markdown
"""
---

### Exercício 5 (Físico-Química): Gráfico de Dispersão com Linha de Tendência (Opcional/Desafio)

**Cenário:** Relação entre a temperatura e a solubilidade de um sal em água.

**Dados:**
* `temperatura_c = np.array([10, 20, 30, 40, 50, 60])`
* `solubilidade_g_100ml = np.array([15, 22, 30, 40, 52, 65])`

**Tarefa:**
1.  Crie um gráfico de dispersão dos dados.
2.  Adicione um título e rótulos aos eixos.
3.  **Desafio:** Adicione uma linha de tendência linear aos dados.
    * **Dica:** Usem `np.polyfit(x, y, grau)` para calcular os coeficientes de um polinómio (grau 1 para uma linha reta) e `np.poly1d(coeficientes)` para criar a função do polinómio. Depois, plotem essa função.
"""

# Célula de Código (para o Exercício 5)
temperatura_c = np.array([10, 20, 30, 40, 50, 60])
solubilidade_g_100ml = np.array([15, 22, 30, 40, 52, 65])

plt.figure(figsize=(10, 6))
plt.scatter(temperatura_c, solubilidade_g_100ml, color='blue', label='Dados de Solubilidade')

# Desafio: Adicionar linha de tendência
# Calcula os coeficientes da linha (grau 1)
coeficientes = np.polyfit(temperatura_c, solubilidade_g_100ml, 1)
# Cria uma função polinomial a partir dos coeficientes
linha_tendencia = np.poly1d(coeficientes)

# Gera pontos para a linha de tendência ao longo do mesmo intervalo de temperatura
x_tendencia = np.linspace(temperatura_c.min(), temperatura_c.max(), 100)
y_tendencia = linha_tendencia(x_tendencia)

plt.plot(x_tendencia, y_tendencia, color='red', linestyle='--', label='Linha de Tendência')

plt.title("Solubilidade de um Sal vs. Temperatura")
plt.xlabel("Temperatura (°C)")
plt.ylabel("Solubilidade (g/100mL água)")
plt.legend()
plt.grid(True)
plt.show()
