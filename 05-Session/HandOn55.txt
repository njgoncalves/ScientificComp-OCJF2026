import pandas as pd
import numpy as np
import io # Para simular leitura de CSV

# Célula de Markdown
"""
---

### Exercício 1 (Físico-Química): Análise de Dados de Reação

**Cenário:** Um cientista registou a concentração de um produto de reação ao longo do tempo e a temperatura da reação.

**Dados (simulados como CSV):**
```csv
Tempo_min,Temperatura_C,Concentracao_Produto_mM
0,25.0,0.0
10,25.5,5.2
20,26.0,9.8
30,26.2,14.5
40,26.5,18.9
50,26.8,22.0
60,27.0,24.5
```

**Tarefa:**
1.  Carregue os dados acima para um DataFrame Pandas a partir da string CSV.
2.  Imprima as primeiras 3 linhas do DataFrame.
3.  Imprima as informações do DataFrame (`.info()`) para verificar os tipos de dados.
4.  Calcule a concentração média do produto.
5.  Encontre o tempo e a concentração máxima registada.
6.  Filtre o DataFrame para mostrar apenas as linhas onde a `Concentracao_Produto_mM` é superior a `15.0`.
"""

# Célula de Código (para o Exercício 1)
csv_reacao = """Tempo_min,Temperatura_C,Concentracao_Produto_mM
0,25.0,0.0
10,25.5,5.2
20,26.0,9.8
30,26.2,14.5
40,26.5,18.9
50,26.8,22.0
60,27.0,24.5
"""

# 1. Carregar os dados
df_reacao = pd.read_csv(io.StringIO(csv_reacao))
print("DataFrame de Reação:\n", df_reacao)

# 2. Primeiras 3 linhas
print("\nPrimeiras 3 linhas:\n", df_reacao.head(3))

# 3. Informações do DataFrame
print("\nInformações do DataFrame:")
df_reacao.info()

# 4. Concentração média
media_concentracao = df_reacao['Concentracao_Produto_mM'].mean()
print(f"\nConcentração média do produto: {media_concentracao:.2f} mM")

# 5. Tempo e concentração máxima
max_concentracao = df_reacao['Concentracao_Produto_mM'].max()
# Para encontrar a linha completa com a concentração máxima:
linha_max_concentracao = df_reacao[df_reacao['Concentracao_Produto_mM'] == max_concentracao]
print(f"\nConcentração máxima do produto: {max_concentracao} mM")
print("Linha com concentração máxima:\n", linha_max_concentracao)


# 6. Filtrar por concentração > 15.0
df_filtrado_concentracao = df_reacao[df_reacao['Concentracao_Produto_mM'] > 15.0]
print("\nDados com Concentração_Produto_mM > 15.0:\n", df_filtrado_concentracao)


# Célula de Markdown
"""
---

### Exercício 2 (Biologia e Geologia): Análise de Dados de Espécies

**Cenário:** Um biólogo registou o número de indivíduos de diferentes espécies em várias áreas de um ecossistema.

**Dados (simulados como dicionário):**
```python
dados_especies = {
    'Area': ['Floresta', 'Floresta', 'Floresta', 'Campo', 'Campo', 'Rio', 'Rio'],
    'Especie': ['Veado', 'Raposa', 'Coelho', 'Coelho', 'Pássaro', 'Peixe', 'Rã'],
    'Num_Individuos': [10, 3, 25, 30, 15, 50, 20]
}
```

**Tarefa:**
1.  Crie um DataFrame a partir do dicionário `dados_especies`.
2.  Calcule o número total de indivíduos por `Area`.
3.  Calcule a média de indivíduos por `Especie`.
4.  Encontre a `Especie` com o maior número de indivíduos total.
5.  Filtre o DataFrame para mostrar apenas as espécies na `Area` 'Floresta' que têm mais de 5 indivíduos.
"""

# Célula de Código (para o Exercício 2)
dados_especies = {
    'Area': ['Floresta', 'Floresta', 'Floresta', 'Campo', 'Campo', 'Rio', 'Rio'],
    'Especie': ['Veado', 'Raposa', 'Coelho', 'Coelho', 'Pássaro', 'Peixe', 'Rã'],
    'Num_Individuos': [10, 3, 25, 30, 15, 50, 20]
}

# 1. Criar DataFrame
df_especies = pd.DataFrame(dados_especies)
print("DataFrame de Espécies:\n", df_especies)

# 2. Número total de indivíduos por Área
total_individuos_por_area = df_especies.groupby('Area')['Num_Individuos'].sum()
print("\nNúmero total de indivíduos por Área:\n", total_individuos_por_area)

# 3. Média de indivíduos por Espécie
media_individuos_por_especie = df_especies.groupby('Especie')['Num_Individuos'].mean()
print("\nMédia de indivíduos por Espécie:\n", media_individuos_por_especie)

# 4. Espécie com o maior número de indivíduos total
# Primeiro, somamos por espécie, depois encontramos o máximo
especie_com_mais_individuos = df_especies.groupby('Especie')['Num_Individuos'].sum().idxmax()
print(f"\nEspécie com o maior número total de indivíduos: {especie_com_mais_individuos}")

# 5. Filtrar espécies na 'Floresta' com mais de 5 indivíduos
df_floresta_filtrado = df_especies[(df_especies['Area'] == 'Floresta') & (df_especies['Num_Individuos'] > 5)]
print("\nEspécies na Floresta com mais de 5 indivíduos:\n", df_floresta_filtrado)

"""
---

### Exercício 3 (Matemática): Análise de Desempenho de Testes

**Cenário:** Um professor registou as notas de vários alunos em diferentes testes.

**Dados (simulados como lista de dicionários):**
```python
dados_testes = [
    {'Aluno': 'Pedro', 'Teste': 'Matemática I', 'Nota': 15},
    {'Aluno': 'Pedro', 'Teste': 'Física I', 'Nota': 12},
    {'Aluno': 'Maria', 'Teste': 'Matemática I', 'Nota': 18},
    {'Aluno': 'Maria', 'Teste': 'Física I', 'Nota': 16},
    {'Aluno': 'João', 'Teste': 'Matemática I', 'Nota': 9},
    {'Aluno': 'João', 'Teste': 'Física I', 'Nota': 10}
]
```

**Tarefa:**
1.  Crie um DataFrame a partir da lista de dicionários `dados_testes`.
2.  Calcule a nota média de cada `Aluno`.
3.  Calcule a nota média para cada `Teste`.
4.  Encontre a nota mínima e máxima de todos os testes.
5.  Identifique quais alunos reprovaram em algum teste (nota < 10).
"""

# Célula de Código (para o Exercício 3)
dados_testes = [
    {'Aluno': 'Pedro', 'Teste': 'Matemática I', 'Nota': 15},
    {'Aluno': 'Pedro', 'Teste': 'Física I', 'Nota': 12},
    {'Aluno': 'Maria', 'Teste': 'Matemática I', 'Nota': 18},
    {'Aluno': 'Maria', 'Teste': 'Física I', 'Nota': 16},
    {'Aluno': 'João', 'Teste': 'Matemática I', 'Nota': 9},
    {'Aluno': 'João', 'Teste': 'Física I', 'Nota': 10}
]

# 1. Criar DataFrame
df_testes = pd.DataFrame(dados_testes)
print("DataFrame de Testes:\n", df_testes)

# 2. Nota média de cada Aluno
media_por_aluno = df_testes.groupby('Aluno')['Nota'].mean()
print("\nNota média por Aluno:\n", media_por_aluno)

# 3. Nota média para cada Teste
media_por_teste = df_testes.groupby('Teste')['Nota'].mean()
print("\nNota média por Teste:\n", media_por_teste)

# 4. Nota mínima e máxima de todos os testes
nota_min_geral = df_testes['Nota'].min()
nota_max_geral = df_testes['Nota'].max()
print(f"\nNota mínima geral: {nota_min_geral}")
print(f"Nota máxima geral: {nota_max_geral}")

# 5. Identificar alunos que reprovaram em algum teste (nota < 10)
alunos_reprovados_df = df_testes[df_testes['Nota'] < 10]
# Para obter apenas os nomes únicos dos alunos que reprovaram
nomes_alunos_reprovados = alunos_reprovados_df['Aluno'].unique()
print(f"\nAlunos que reprovaram em algum teste: {nomes_alunos_reprovados}")
